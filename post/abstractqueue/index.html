AbstractQueue

2020-11-13 2 min read # java # collection # queue

AbstractQueue抽象类，实现了Queue接口的部分通用方法。

public abstract class AbstractQueue<E>
    extends AbstractCollection<E>
    implements Queue<E> {}

  1. add()添加

    //add方法，基于 offer方法。成功返回true,否则，抛出异常
public boolean add(E e) {
    if (offer(e))
        return true;
    else
        throw new IllegalStateException("Queue full");
}

  2. remove() 取出元素

    //调用poll，成功返回元素，否则抛出异常。
public E remove() {
    E x = poll();
    if (x != null)
        return x;
    else
        throw new NoSuchElementException();
}

  3. element()

    public E element() {
        E x = peek();
        if (x != null)
            return x;
        else
            throw new NoSuchElementException();
    }

  4. clear()

    //优秀啊！poll 元素，直到返回空。代表清空了元素。但是有线程安全问题吗？可能会存在哦。~~~~
public void clear() {
        while (poll() != null);
    }

  5. addAll()

    //循环调用add方法，一个一个元素添加到集合中，直到集合全部添加完成。but， 
// c==this,throw new IllegalArgumentException();？为什么要设计成自己不能添加自己
  public boolean addAll(Collection<? extends E> c) {
        if (c == null)
            throw new NullPointerException();
        if (c == this)
            throw new IllegalArgumentException();
        boolean modified = false;
        for (E e : c)
            if (add(e))
                modified = true;
        return modified;
    }
